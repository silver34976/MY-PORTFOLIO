const totalPhotos = photos.length; const photosPerPage = 5; let currentIndex = 0; const gallery = document.getElementById('gallery'); function showPhotos() { gallery.innerHTML = ''; for(let i = 0; i < photosPerPage; i++) { const index = (currentIndex + i) % totalPhotos; const img = document.createElement('img'); img.src = photos[index]; img.alt = `Photo ${index + 1}`; gallery.appendChild(img); } currentIndex = (currentIndex + photosPerPage) % totalPhotos; } showPhotos(); setInterval(showPhotos, 10000); // 10 seconds
